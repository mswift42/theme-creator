def fac(num):
"return factorial of number num"
    if num == 0:
        return 1
    else:
        return num * fac(num-1)

def sumDigits(num):
  "return sum of all digits in number num"
  s = str(num) # convert number to string of its digits.
  total = 0
  for i in s:
      total += int(i) # add int value of each digit to total.
  return total

print(sumDigits(fac(100)))
